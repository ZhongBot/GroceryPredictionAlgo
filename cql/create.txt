CREATE TABLE fydp.products (
    product_id int,
	price double,
    avg_price double,
    category text,
    brand text,
	primary key (product_id)
);

CREATE TABLE fydp.inventory (
    user_id text,
	product_id int,
	inventory counter,
	primary key (user_id, product_id)
);

CREATE TABLE fydp.satisfaction (
    user_id text,
	product_id int,
	s_ema double,
	primary key (user_id, product_id)
);

CREATE TABLE fydp.satisfaction_current (
    user_id text,
	product_id int,
	rating double,
	primary key (user_id, product_id)
);

CREATE TABLE fydp.consumption (
    user_id text,
	product_id int,
	days_elapsed int,
	quantity_purchased int,
	primary key (user_id, product_id)
);

CREATE TABLE fydp.consumption_freq (
    user_id text,
	product_id int,
	date timestamp,
	primary key (user_id, product_id, date)
);

CREATE TABLE fydp.purchase_ind (
    user_id text,
	product_id int,
	compute_time timestamp,
	purchase_ind double,
	primary key (user_id, product_id, compute_time)
);

CREATE TABLE fydp.shoppinglists (
    sl_id uuid PRIMARY KEY,
    autogenerate boolean,
    name text,
    products list<int>,
    time timestamp,
    userid text
);


CREATE TABLE fydp.users (
    userid text PRIMARY KEY,
    name text,
    password text,
    phone text
);

CREATE TABLE fydp.products (
    product_id int PRIMARY KEY,
    avg_price double,
    brand text,
    category text,
    price double
);
